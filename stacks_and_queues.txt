1. Stack uses LIFO, queue uses FIFO

2. They are both used to manage data, but in different orders.

3. I would use LIFO for both, since the last thing you did(or undid) would be the first thing you'd like to undo(or redo).

var word = "Bloc!"

function mirror(original) {
    var newStack = [];
    var tempStack = original.split('');
    var lengthStack = tempStack.length;
    for (i=0; i < lengthStack; i++){
        var next = tempStack.pop();
        newStack.push(next);
    }
}

mirror(word);

var queue = ['hi']

function queue(element) {
    queue.push(element)
}

function dequeue() {
    var temp = queue[0];
    var tempIndex = queue.indexOf(temp)
    queue.splice(tempIndex,1)
}
